def preprocess_str(raw_str):
    # Convert to lowercase
    text = raw_str.lower()
    
    # Tokenize and remove punctuation
    tokens = nltk.word_tokenize(text.translate(str.maketrans('', '', string.punctuation)))
    stemmer = SnowballStemmer("english")
    
    # Filter stopwords, non-alpha, and words less than 4 characters
    processed_text = [stemmer.stem(word) for word in tokens if word not in all_stopwords and word.isalpha() and len(word) >= 4]
    
    # Join tokens back into a string
    return ' '.join(processed_text)
